# 1. Definición de variables con diferentes tipos de datos
nombre = "Camilo"  # Variable de tipo Cadena (string)
edad = 28          # Variable de tipo Entero (int)

# 2. Operación con variables
edad_en_10_anios = edad + 10

# 3. Impresión del resultado utilizando una f-string (cadena formateada)
mensaje = f"Hola, me llamo {nombre}. Tengo {edad} años y mido {estatura} metros. En 10 años, tendré {edad_en_10_anios} años."

print(mensaje)
